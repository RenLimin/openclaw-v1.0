# 🤖 多模型适配抽象层与智能路由 v1.0

> **版本**: v1.0
> **发布日期**: 2026-05-09
> **核心目标**: 模型无关、自动路由、故障转移、成本优化
> **设计原则**: 一次编写，任意模型运行；统一接口，屏蔽差异

---

## 🎯 解决的核心问题

### 痛点场景
```
❌ 问题1：代码与具体模型绑定，切换模型需要大规模改代码
❌ 问题2：不同模型API参数/返回格式不统一，适配成本高
❌ 问题3：某个模型服务挂了，整个系统跟着挂，没有故障转移
❌ 问题4：简单任务用大模型，成本高速度慢，资源浪费
❌ 问题5：不同模型各有所长，无法发挥组合优势
```

## 📋 标准统一接口定义

### 聊天补全接口（所有模型通用）

```python
from abc import ABC, abstractmethod
from typing import List, Dict, Any, Optional, Iterator
from dataclasses import dataclass

@dataclass
class ChatMessage:
    """统一消息格式"""
    role: str  # system/user/assistant/tool
    content: str
    name: Optional[str] = None
    tool_call_id: Optional[str] = None

@dataclass
class FunctionDefinition:
    """统一函数定义格式"""
    name: str
    description: str
    parameters: Dict[str, Any]

@dataclass
class ChatCompletionRequest:
    """统一请求格式"""
    model: Optional[str] = None  # None表示交给智能路由选择
    messages: List[ChatMessage] = None
    functions: Optional[List[FunctionDefinition]] = None
    function_call: Optional[str] = None
    temperature: float = 0.7
    top_p: float = 1.0
    max_tokens: Optional[int] = None
    stream: bool = False
    timeout: int = 60

@dataclass
class ChatCompletionChoice:
    """统一返回格式"""
    index: int
    message: ChatMessage
    finish_reason: str  # stop/length/function_call/content_filter
    usage: Dict[str, int]  # prompt_tokens, completion_tokens, total_tokens

@dataclass
class ChatCompletionResponse:
    """统一响应格式"""
    id: str
    model: str
    provider: str
    choices: List[ChatCompletionChoice]
    usage: Dict[str, int]
    latency_ms: int
    raw_response: Optional[Any] = None  # 原始响应，debug用

class BaseModelProvider(ABC):
    """所有模型提供者的基类"""

    @abstractmethod
    async def chat_completion(
        self, request: ChatCompletionRequest
    ) -> ChatCompletionResponse:
        """聊天补全 - 阻塞模式"""
        pass

    @abstractmethod
    async def chat_completion_stream(
        self, request: ChatCompletionRequest
    ) -> Iterator[ChatCompletionResponse]:
        """聊天补全 - 流式模式"""
        pass

    @abstractmethod
    async def embedding(self, texts: List[str]) -> List[List[float]]:
        """向量生成"""
        pass

    @abstractmethod
    async def health_check(self) -> bool:
        """健康检查"""
        pass

    @abstractmethod
    def get_model_info(self) -> Dict[str, Any]:
        """获取模型信息"""
        return {
            "provider": self.provider_name,
            "model": self.model_name,
            "max_context_length": self.max_context,
            "max_output_tokens": self.max_output,
            "cost_per_1k_input_tokens": self.input_cost,
            "cost_per_1k_output_tokens": self.output_cost,
            "supported_features": self.features
        }
```

```python
class OpenAIProvider(BaseModelProvider):
    """OpenAI 模型适配层"""

    async def chat_completion(self, request: ChatCompletionRequest):
        # 1. 标准请求 → OpenAI格式转换
        openai_messages = self._convert_to_openai_messages(request.messages)
        openai_functions = self._convert_to_openai_functions(request.functions)

        # 2. 调用OpenAI API
        try:
            response = await self.client.chat.completions.create(
                model=request.model or "gpt-4o",
                messages=openai_messages,
                functions=openai_functions,
                temperature=request.temperature,
                max_tokens=request.max_tokens,
                timeout=request.timeout
            )
        except Exception as e:
            # 3. 统一错误映射
            raise self._map_openai_error(e)

        # 4. OpenAI响应 → 标准格式转换
        return self._convert_to_standard_response(response, request)
```

### 多维路由决策因素

```python
async def intelligent_route(request: ChatCompletionRequest):
    """智能路由核心算法"""

    # 1. 任务复杂度评估
    task_complexity = evaluate_task_complexity(request)
    # 结果: simple/medium/complex/very_complex

    # 2. 能力需求分析
    requirements = analyze_requirements(request)
    # 结果: {
    #   "needs_function_call": True/False,
    #   "needs_long_context": True/False,
    #   "needs_code_generation": True/False,
    #   "needs_reasoning": True/False,
    #   "needs_multimodal": True/False
    # }

    # 3. 筛选能满足需求的候选模型
    candidate_models = filter_models_by_capability(requirements)

    # 4. 健康状态过滤（排除故障中的模型）
    healthy_models = filter_healthy_models(candidate_models)

    # 5. 成本最优排序（相同能力选最便宜）
    cost_ranked = sort_by_cost(healthy_models)

    # 6. 延迟最优排序（考虑当前负载）
    latency_ranked = sort_by_latency(cost_ranked)

    # 7. 选择最佳模型
    best_model = latency_ranked[0]

    # 8. 记录路由决策，用于后续优化
    record_routing_decision(request, best_model)

    return best_model
```

## 🔄 故障转移与自动重试

### 三级故障转移策略

```python
async def call_with_fallback(request: ChatCompletionRequest):
    """带故障转移的模型调用"""

    # 1. 首选模型列表（按优先级排序）
    preferred_models = get_preferred_models(request)

    for i, model in enumerate(preferred_models):
        try:
            # 2. 尝试调用当前模型
            result = await model.chat_completion(request)
            record_success(model, request)
            return result

        except RateLimitError:
            # 3. 限流错误：立即切换下一个模型（不要重试同一个）
            record_warning(f"模型 {model} 限流，切换下一个")
            continue

        except ServiceUnavailableError:
            # 4. 服务不可用：标记故障，切换下一个
            mark_model_unhealthy(model, duration_seconds=300)
            record_error(f"模型 {model} 不可用，已标记为不健康")
            continue

        except TimeoutError:
            # 5. 超时错误：重试2次，然后切换
            if i < 2:
                retry_delay = calculate_exponential_backoff(i)
                await asyncio.sleep(retry_delay)
                continue  # 重试同一个
            else:
                mark_model_unhealthy(model, duration_seconds=60)
                continue  # 切换下一个

        except Exception as e:
            # 6. 其他错误：记录日志，切换下一个
            record_error(f"模型 {model} 未知错误: {e}")
            continue

    # 7. 所有模型都失败
    raise AllModelsFailedError("所有可用模型都调用失败")
```

### 智能降级策略

```python
def optimize_for_cost(request: ChatCompletionRequest):
    """成本优化策略"""

    # 1. 非高峰时段降级（0:00-8:00用便宜模型）
    if is_off_peak_hours():
        if request.complexity == 'medium':
            return downgrade_model(request, 'gpt-3.5-turbo')

    # 2. 成本预算监控
    if get_monthly_cost() > get_monthly_budget() * 0.8:
        # 预算超80%，全部非核心任务降级
        if not is_core_task(request):
            return downgrade_to_cheapest_model(request)

    # 3. 长上下文优化（超过32K用Claude，比GPT-4便宜50%）
    if get_token_count(request) > 32000:
        return prefer_model(request, 'claude-3-sonnet')

    # 4. 批量任务使用便宜模型
    if is_batch_task(request) and len(request.batch) > 10:
        return prefer_model(request, 'gpt-3.5-turbo')

    return request
```

```python
async def evaluate_model_quality(
    model_a: str,
    model_b: str,
    test_cases: List[ChatCompletionRequest]
):
    """A/B测试比较两个模型质量"""

    results = {
        "model_a": {"score": 0, "wins": 0, "losses": 0, "ties": 0},
        "model_b": {"score": 0, "wins": 0, "losses": 0, "ties": 0}
    }

    for test_case in test_cases:
        # 并行调用两个模型
        result_a, result_b = await asyncio.gather(
            call_model(model_a, test_case),
            call_model(model_b, test_case)
        )

        # 自动评估（也可以引入人类评估）
        winner = auto_evaluate_result_quality(
            test_case, result_a, result_b
        )

        if winner == 'A':
            results['model_a']['wins'] += 1
        elif winner == 'B':
            results['model_b']['wins'] += 1
        else:
            results['model_a']['ties'] += 1
            results['model_b']['ties'] += 1

    return results
```
